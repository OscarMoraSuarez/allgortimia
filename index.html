    let tiempoHora=2;

    function calcularMinutos(distancia,tiempoHoras,tiempoMinutos){
    let minutos=tiempoHoras*60;
    let minutosTotales=minutos+tiempoMinutos;
    let minutosKilometro=minutosTotales/distancia;
